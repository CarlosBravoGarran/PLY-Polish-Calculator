# Pruebas básicas de operadores
+ 3 4
- 10 5
* 2 8
/ 20 4

# Pruebas con números flotantes
+ 1.5 2.5
- 10.0 3.5
* 2.3 4.1
/ 9.0 3.0

# Pruebas con números binarios
+ 0b1010 0b0101
- 0b1111 0b0001
* 0b100 0b011
/ 0b1100 0b0100

# Pruebas con números hexadecimales (deben ser en mayúscula)
+ 0xA 0x5
- 0xF 0x1
* 0x4 0x3
/ 0xC 0x4

# Pruebas con operadores científicos
exp 1
log 10
sin 0
cos 3.1416

# Pruebas con el operador unario neg
neg 5
neg 0b1010
neg 0xA
neg 3.14

# Pruebas combinadas
+ neg 5 3
- 10 neg 2
* neg 4 2
/ 20 neg 4

# Pruebas con comentarios
# Este comentario debe ser ignorado
+ 3 3  # Esto es una suma con comentario

# Pruebas con casos extremos
/ 1 0  # División por cero, debería dar NaN
log neg 1 # Logaritmo de número negativo, debería dar NaN
